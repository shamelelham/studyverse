/* SIGNUP */
function signup() {

  let userName =
    document.getElementById("name").value.trim();

  let userEmail =
    document.getElementById("email").value.trim();

  let userPassword =
    document.getElementById("password").value.trim();

  let userRole =
    document.getElementById("role").value;

  if (
    userName === "" ||
    userEmail === "" ||
    userPassword === ""
  ) {
    alert("⚠️ Please fill in all fields!");
    return;
  }

  if (localStorage.getItem(userEmail)) {
    alert("⚠️ Account already exists!");
    return;
  }

  let user = {
    name:userName,
    email:userEmail,
    password:userPassword,
    role:userRole
  };

  localStorage.setItem(
    userEmail,
    JSON.stringify(user)
  );

  alert("✅ Registration successful!");

  showLogin();
}

/* LOGIN */
function login() {

  let userEmail =
    document.getElementById("email").value.trim();

  let userPassword =
    document.getElementById("password").value.trim();

  if (
    userEmail === "" ||
    userPassword === ""
  ) {
    alert("⚠️ Please enter email and password!");
    return;
  }

  let user =
    JSON.parse(localStorage.getItem(userEmail));

  if (!user) {
    alert("⚠️ User not found!");
    return;
  }

  if (user.password !== userPassword) {
    alert("⚠️ Incorrect password!");
    return;
  }

  currentUser = user;

  showApp();
}

/* RESET PASSWORD */
function resetPassword() {

  let emailInput =
    document.getElementById("resetEmail").value.trim();

  let newPassInput =
    document.getElementById("newPassword").value.trim();

  if (
    emailInput === "" ||
    newPassInput === ""
  ) {
    alert("⚠️ Please fill in all fields!");
    return;
  }

  let user =
    JSON.parse(localStorage.getItem(emailInput));

  if (!user) {
    alert("⚠️ User not found!");
    return;
  }

  user.password = newPassInput;

  localStorage.setItem(
    user.email,
    JSON.stringify(user)
  );

  alert("✅ Password updated successfully!");

  showLogin();
}

      .join("")
      .toUpperCase();

  document
    .getElementById("userInitials")
    .innerText = initials;

  loadPage("home");
}

/* SAVE PROFILE */
function saveProfile() {

  let updatedName =
    document.getElementById("newName")
    .value
    .trim();

  if (updatedName === "") {
    alert("⚠️ Name cannot be empty!");
    return;
  }

  currentUser.name = updatedName;

  localStorage.setItem(
    currentUser.email,
    JSON.stringify(currentUser)
  );

  let initials =
    currentUser.name
      .split(" ")
      .map(n => n[0])
      .join("")
      .toUpperCase();

  document
    .getElementById("userInitials")
    .innerText = initials;

  alert("✅ Profile updated successfully!");

  loadPage("profile");
}
